189def compile(strings, black='X', white='.',xor='o'): 

190 """pygame.cursors.compile(strings, black, white,xor) -> data, mask 

191compile cursor strings into cursor data 

192 

193This takes a set of strings with equal length and computes 

194the binary data for that cursor. The string widths must be 

195divisible by 8. 

196 

197The black and white arguments are single letter strings that 

198tells which characters will represent black pixels, and which 

199characters represent white pixels. All other characters are 

200considered clear. 

201 

202This returns a tuple containing the cursor data and cursor mask 

203data. Both these arguments are used when setting a cursor with 

204pygame.mouse.set_cursor(). 

205""" 

206 

207 #first check for consistent lengths 

208 size = len(strings[0]), len(strings) 

209 if size[0] % 8 or size[1] % 8: 

210 raise ValueError("cursor string sizes must be divisible by 8 %s" % 

211 size) 

212 for s in strings[1:]: 

213 if len(s) != size[0]: 

214 raise ValueError("Cursor strings are inconsistent lengths") 

215 

216 #create the data arrays. 

217 #this could stand a little optimizing 

218 maskdata = [] 

219 filldata = [] 

220 maskitem = fillitem = 0 

221 step = 8 

222 for s in strings: 

223 for c in s: 

224 maskitem = maskitem << 1 

225 fillitem = fillitem << 1 

226 step = step - 1 

227 if c == black: 

228 maskitem = maskitem | 1 

229 fillitem = fillitem | 1 

230 elif c == white: 

231 maskitem = maskitem | 1 

232 elif c == xor: 

233 fillitem = fillitem | 1 

234 if not step: 

235 maskdata.append(maskitem) 

236 filldata.append(fillitem) 

237 maskitem = fillitem = 0 

238 step = 8 

239 return tuple(filldata), tuple(maskdata)

244def load_xbm(curs, mask): 

245 """pygame.cursors.load_xbm(cursorfile, maskfile) -> cursor_args 

246reads a pair of XBM files into set_cursor arguments 

247 

248Arguments can either be filenames or filelike objects 

249with the readlines method. Not largely tested, but 

250should work with typical XBM files. 

251""" 

252 def bitswap(num): 

253 val = 0 

254 for x in range(8): 

255 b = num&(1<<x) != 0 

256 val = val<<1 | b 

257 return val 

258 

259 if type(curs) is type(''): 

260 with open(curs) as cursor_f: 

261 curs = cursor_f.readlines() 

262 else: 

263 curs = curs.readlines() 

264 

265 if type(mask) is type(''): 

266 with open(mask) as mask_f: 

267 mask = mask_f.readlines() 

268 else: 

269 mask = mask.readlines() 

270 

271 #avoid comments 

272 for line in range(len(curs)): 

273 if curs[line].startswith("#define"): 

274 curs = curs[line:] 

275 break 

276 for line in range(len(mask)): 

277 if mask[line].startswith("#define"): 

278 mask = mask[line:] 

279 break 

280 #load width,height 

281 width = int(curs[0].split()[-1]) 

282 height = int(curs[1].split()[-1]) 

283 #load hotspot position 

284 if curs[2].startswith('#define'): 

285 hotx = int(curs[2].split()[-1]) 

286 hoty = int(curs[3].split()[-1]) 

287 else: 

288 hotx = hoty = 0 

289 

290 info = width, height, hotx, hoty 

291 

292 for line in range(len(curs)): 

293 if curs[line].startswith('static char') or curs[line].startswith('static unsigned char'): 

294 break 

295 data = ' '.join(curs[line+1:]).replace('};', '').replace(',', ' ') 

296 cursdata = [] 

297 for x in data.split(): 

298 cursdata.append(bitswap(int(x, 16))) 

299 cursdata = tuple(cursdata) 

300 

301 for line in range(len(mask)): 

302 if mask[line].startswith('static char') or mask[line].startswith('static unsigned char'): 

303 break 

304 data = ' '.join(mask[line+1:]).replace('};', '').replace(',', ' ') 

305 maskdata = [] 

306 for x in data.split(): 

307 maskdata.append(bitswap(int(x, 16))) 

308 maskdata = tuple(maskdata) 

309 return info[:2], info[2:], cursdata, maskdata
